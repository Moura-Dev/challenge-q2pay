Table users as U {
  id int [pk, increment] // auto-increment
  full_name varchar
  cpfcnpj varchar
  email  varchar
  login varchar
  password varchar
  active boolean
  created_at datetime
  updated_at datetime
}


Table wallet as W {
  id int [pk, increment]
  balance decimal
  user_id int
  version int
}

Table transaction as T {
  id int [pk, increment]
  value decimal
  payer int
  payee int
  created_at datetime

}

Ref: U.id > W.user_id
Ref: T.payer > U.id
Ref: T.payee > U.id
